from gturtle import *
def quadrat(seitenlaenge):
    repeat 4:
        forward(seitenlaenge)
        right(90)
        
def quadrat_muster(aktuelle_seitenlaenge):
    repeat 50:
        quadrat(aktuelle_seitenlaenge)
        aktuelle_seitenlaenge += 3
        
makeTurtle()
hideTurtle()
quadrat_muster(30)
\end{lstlisting}
