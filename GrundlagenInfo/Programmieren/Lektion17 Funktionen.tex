\begin{frame}{Fazit}
\begin{itemize}[<+->]
	\item Funktionen sind Definitionen...
	\item Die wie ein \textbf{Kochrezept} funktionieren:
	\begin{itemize}
		\item Gewisse Inputs, bzw. \textbf{Parameter} können akzeptiert werden (müssen aber nicht)
		\item Gewisse Outputs, bzw. \textbf{Return-Werte }können zurückgeben werden (müssen aber nicht)
	\end{itemize}
	\item \emoji{thumbs-up} Vorteil von Return-Werten: man kann nun das Resultat einer Funktion in einer Variable speichern und \textbf{weiterverwenden}
\end{itemize}
\end{frame}
