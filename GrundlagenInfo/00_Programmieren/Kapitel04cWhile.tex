\begin{frame}[fragile]{Spirale: Flussgesteuerter Abbruch (\lstinline[basicstyle=\ttfamily]|break|)}
\begin{columns}
\begin{column}{.7\textwidth}
\begin{lstlisting}
from gturtle import *
makeTurtle()

def spirale(seite, add, max_seite):
    repeat: # unendliche Schleife
    	if seite > max_seite:
    	    break
        fd(seite)
        rt(90)
        seite+=add

spirale(10, 10, 100)
\end{lstlisting}

\end{column}
