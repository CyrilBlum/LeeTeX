\begin{frame}[fragile]{while vs. break}
\begin{columns}
\begin{column}{.4\textwidth}
\begin{lstlisting}[escapechar=!,basicstyle=\tiny\ttfamily]
from gturtle import *
makeTurtle()

def spirale(seite, add, max_seite):
    repeat:                !\tikzmark{fromStart}!
    	if seite > max_seite:
    	    break          !\tikzmark{fromEnd}!
        fd(seite)
        rt(90)
        seite+=add

spirale(10, 10, 100)
\end{lstlisting}
\end{column}
\begin{column}{.6\textwidth}
\begin{lstlisting}[escapechar=!,basicstyle=\tiny\ttfamily]
from gturtle import *
makeTurtle()

def spirale(seite, add, max_seite):
    !\tikzmark{toStart}!!\tikzmark{toEnd}!while seite > max_seite:
        fd(seite)
        rt(90)
        seite+=add
        
spirale(10, 10, 100)
\end{lstlisting}

\begin{frame}[fragile]{Wichtige Bemerkungen}
\begin{itemize}[<+->]
\item \lstinline|break| bricht nicht aus \lstinline|def| aus, sondern aus dem (aktuellen) \lstinline|repeat|!
\item Ein \lstinline|repeat:| (unendliche Schleife) ohne \lstinline|break| kann zum Absturz des Programms führen \emoji{face-with-spiral-eyes}
\item Je nachdem kann das auch in einem \lstinline|while| passieren, sofern die Bedingung nach dem \lstinline|while| so geschrieben ist, dass sie immer wahr (\lstinline|True|) ist.
\item \textbf{Speichern Sie regelmässig Ihre Aufgaben!}

\end{itemize}
