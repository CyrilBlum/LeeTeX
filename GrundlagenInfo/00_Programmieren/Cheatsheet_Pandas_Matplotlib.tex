    \cheatSheetSubtitle{Lineare Regression}{Die folgenden Boxen enthalten die wichtigsten Formeln und Befehle für die lineare Regression von Hand und in Python.}

    \begin{cheatSheetBox}[title=Lineare Funktion]
      Formel einer linearen Funktion:
      \[y = m \cdot x + q\]
      Wobei m die Steigung und q den y-Achsenabschnitt darstellt.
      \[m=\frac{y_2 - y_1}{x_2 - x_1},\quad\quad q = y_1 - m \cdot x_1\]
    \end{cheatSheetBox}
    \begin{cheatSheetBox}[title=Lineare Regression]
      \begin{itemize}
        \item \textbf{Unabhängige Variable}: \lstinline|df['Schlaf']|
        \item \textbf{Abhängige Variable}: \lstinline|df['Note']|
      \end{itemize}
      \begin{lstlisting}
model = smf.ols(formula='Note ~ Schlaf', data=df).fit()
print(model.params) # Koeffizienten
# Vorhersage (für alle Daten)
df['Note_pred'] = model.predict(df['Schlaf']) 
# Vorhersage (für einen bestimmten Wert)
pred = model.predict({'Schlaf': [6]})
\end{lstlisting}
    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Multiple lineare Regression]
      Beispiel: Note als Funktion von Lernaufwand und Schlaf (sowie möglicherweise weiteren Variablen).
      \textbf{Formel:}
      \[
        y = a_0 + a_1 x_1 + a_2 x_2 + \ldots + a_n x_n
      \]
      \begin{itemize}
        \item $y$: Zielvariable (z.B. Note)
        \item $x_1, x_2, \ldots, x_n$: unabhängige Variablen (z.B. Lernaufwand, Schlaf, etc.)
        \item $a_0$: Achsenabschnitt (Intercept)
        \item $a_1, a_2, \ldots, a_n$: Regressionskoeffizienten
      \end{itemize}
    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Multiple lineare Regression (in Python)]

    \cheatSheetSubtitle{Markov-Ketten}{
      Die folgenden Boxen enthalten die wichtigsten Formeln und Befehle für Markov-Ketten, von Hand und in Python. Die folgenden Boxen beziehen sich auf das Wetter-Beispiel, bei dem es zwei Zustände gibt: Sonne und Regen. Die Übergangswahrscheinlichkeiten sind:
      \begin{itemize}
        \item Wenn heute \emoji{sun}: Morgen zu 80\% \emoji{sun}, zu 20\% \emoji{umbrella}
        \item Wenn heute \emoji{umbrella}: Morgen zu 40\% \emoji{sun}, zu 60\% \emoji{umbrella}
      \end{itemize}
    }

    \begin{cheatSheetBox}[title=Gerichteter Graph]
      Das Wetter-Beispiel kann als gerichteter Graph dargestellt werden:

      \begin{center}
        \begin{tikzpicture}[node distance=3cm, auto, thick]
          \node[state] (S) {\emoji{sun}};
          \node[state] (R) [right of=S] {\emoji{umbrella}};

          \path[->]
          (S) edge[bend left] node {0.2} (R)
          (S) edge[loop above] node {0.8} (S);

          \path[->]
          (R) edge[bend left] node {0.4} (S)
          (R) edge[loop above] node {0.6} (R);
        \end{tikzpicture}
      \end{center}


      \begin{itemize}[leftmargin=*,labelsep=0.5em]
        \item[{\faCircle[regular]}]  \textbf{Knoten}: Zustände
        \item[\faArrowRight] \textbf{Kanten}: Übergänge mit Wahrscheinlichkeiten
      \end{itemize}


    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Übergangsmatrix]
      Das Wetter-Beispiel kann auch als Übergangsmatrix dargestellt werden. Die Matrix zeigt die Wahrscheinlichkeiten, von einem Zustand in einen anderen zu wechseln. Die Zeilen repräsentieren den aktuellen Zustand, die Spalten den zukünftigen Zustand. Normalerweise wird die Matrix wie rechts (ohne Labels \enquote{Von} und \enquote{Nach}) dargestellt.
      \begin{center}
        \begin{tabular}{cc}
          % Erste Subtabelle: Übergangstabelle mit Labels
          \begin{tabular}{c c|c|c}
                                                          &                  & \multicolumn{2}{c}{\textbf{Nach}}                    \\
                                                          &                  & \emoji{sun}                       & \emoji{umbrella} \\\midrule\midrule
            \multirow{2}{*}{\rotatebox{90}{\textbf{Von}}} & \emoji{sun}      & 0.8                               & 0.2              \\\cmidrule{2-4}
                                                          & \emoji{umbrella} & 0.4                               & 0.6              \\
          \end{tabular}
           &
          % Zweite Subtabelle: Matrixdarstellung ohne Labels
          \(
          P = \begin{pmatrix}
                0.8 & 0.2 \\
                0.4 & 0.6 \\
              \end{pmatrix}
          \)
        \end{tabular}
      \end{center}
      \begin{itemize}[leftmargin=*,labelsep=0.5em]
        \item \textbf{Links}: Übergangstabelle mit Labels
        \item \textbf{Rechts}: Matrixdarstellung ohne Labels
      \end{itemize}
    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Matrix-Multiplikation]
      \[
        \begin{pmatrix}
          \textcolor{red}{a} & \textcolor{blue}{b} \\
        \end{pmatrix}
        \cdot
        \begin{pmatrix}
          \textcolor{teal}{e}   & \textcolor{violet}{f} \\
          \textcolor{orange}{g} & \textcolor{brown}{h}  \\
        \end{pmatrix}
        =
        \begin{pmatrix}
          \textcolor{red}{a} \cdot \textcolor{teal}{e} + \textcolor{blue}{b} \cdot \textcolor{orange}{g} & \textcolor{red}{a} \cdot \textcolor{violet}{f} + \textcolor{blue}{b} \cdot \textcolor{brown}{h} \\
        \end{pmatrix}
      \]
    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Vorhersage von Zuständen]

      Angenommen, heute ist es sonnig. Wir wissen also, dass es zu 100\% sonnig und zu 0\% regnerisch ist. Der Zustandsvektor ist also 100\% (=1) Sonne und 0\% (=0) Regen:
      \[\begin{aligned}
          \pi_0 =
           & \begin{pmatrix}
               1 & 0
             \end{pmatrix}                                                   \\
           & \begin{matrix}
               \;\; \Uparrow & \Uparrow
             \end{matrix}                                         \\
           & \begin{matrix}
               \; \text{\emoji{sun}} & \text{\emoji{umbrella}}
             \end{matrix}
        \end{aligned}\]
      Wir berechnen zuerst $\pi_1$, den Zustand nach einem Tag (mit den Wahrscheinlichkeiten aus der Übergangsmatrix $P$):
      \[
        P = \begin{pmatrix}
          0.8 & 0.2 \\
          0.4 & 0.6 \\
        \end{pmatrix}
      \]

      \[\begin{aligned}
          \pi_1 & = \pi_0 \cdot P                               \\
          \pi_1 & =
          \begin{pmatrix}
            1 & 0 \\
          \end{pmatrix}
          \cdot
          \begin{pmatrix}
            0.8 & 0.2 \\
            0.4 & 0.6 \\
          \end{pmatrix}                                        \\
          \pi_1 & =
          \begin{pmatrix}
            1 \cdot 0.8 + 0 \cdot 0.4 & 1 \cdot 0.2 + 0 \cdot 0.6 \\
          \end{pmatrix} \\
          \pi_1 & =
          \begin{pmatrix}
            0.8 + 0 & 0.2 + 0 \\
          \end{pmatrix}                                     \\
          \pi_1 & =
          \underline{
              \begin{pmatrix}
                0.8 & 0.2 \\
              \end{pmatrix}
            }
        \end{aligned}\]

      Dabei bezeichnet \underline{0.8} die Wahrscheinlichkeit, dass es morgen sonnig ist, und \underline{0.2} die Wahrscheinlichkeit, dass es morgen regnet.

      Dann berechnen wir den Zustand nach zwei Tagen:

      \[
        \begin{aligned}
          \pi_2 & = \pi_1 \cdot P                                       \\
          \pi_2 & =
          \begin{pmatrix}
            0.8 & 0.2 \\
          \end{pmatrix}
          \cdot
          \begin{pmatrix}
            0.8 & 0.2 \\
            0.4 & 0.6 \\
          \end{pmatrix}                                                \\
          \pi_2 & =
          \begin{pmatrix}
            0.8 \cdot 0.8 + 0.2 \cdot 0.4 & 0.8 \cdot 0.2 + 0.2 \cdot 0.6 \\
          \end{pmatrix} \\
          \pi_2 & =
          \begin{pmatrix}
            0.64 + 0.08 & 0.16 + 0.12 \\
          \end{pmatrix}                                     \\
          \pi_2 & =
          \underline{\underline{
              \begin{pmatrix}
                0.72 & 0.28 \\
              \end{pmatrix}
            }}
        \end{aligned}
      \]

      Die Wahrscheinlichkeit, nach zwei Tagen Sonne zu haben, beträgt \underline{\underline{72\%}}. Die Wahrscheinlichkeit, nach zwei Tagen Regen zu haben, beträgt \underline{\underline{28\%}}.
    \end{cheatSheetBox}

    \begin{cheatSheetBox}[title=Python: Erstellung der Übergangsmatrix]
      \begin{lstlisting}
# Zustände: 0 = Sonne, 1 = Regen
P = np.array([
    [0.8, 0.2],
    [0.4, 0.6]
])
\end{lstlisting}
    \end{cheatSheetBox}
    \begin{cheatSheetBox}[title=Python: Matrixmultiplikation / Vorhersage]
      \begin{lstlisting}
# Startzustand: 100% Sonne
pi0 = np.array([1, 0])

# Nach einem Tag
pi1 = pi0 @ P
print("Nach 1 Tag:", pi1)

# Nach zwei Tagen
pi2 = pi1 @ P
print("Nach 2 Tagen:", pi2)
\end{lstlisting}
    \end{cheatSheetBox}
    \begin{cheatSheetBox}[title=Python: Vorhersage (Zufallsimulation)]
      \begin{lstlisting}[numbers=left]
zustand = "Sonne"    # Anfangszustand
history = [zustand]  # Liste simulierter Zustände
schritte = 20        # Anzahl Simulationsschritte

for i in range(schritte):
    if zustand == "Sonne":
        wahrscheinlichkeit = P[0]
    else:
        wahrscheinlichkeit = P[1]

    zustand = random.choices(["Sonne", "Regen"], weights=wahrscheinlichkeit).pop()
    history.append(zustand)

print(history)
\end{lstlisting}
    \end{cheatSheetBox}
