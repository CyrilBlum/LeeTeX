\section{Division mit Rest}\label{sec:DivisionMitRest}
\begin{mydefinition}[Kongruenz ganzer Zahlen]
Sei $m > 0$ eine fest gewählte ganze Zahl. Seien $a$ und $b$ ganze Zahlen. Dann heissen $a$ und $b$ \textbf{kongruent modulo}\index{Kongruenz modulo} $m$, geschrieben
\begin{align*}
	a \equiv b \pmod{m},
\end{align*}
fall $m$ die Differenz $(a - b)$ teilt.
\end{mydefinition}

\begin{mydefinition}[Modulo-Operation]
Sei $m > 0$ eine fest gewählte ganze Zahl und $a$ eine ganze Zahl. Offensichtlich ist $a$ kongruent modulo $m$ zu genau einer Zahl $b$ mit $b\in\lrc{0, 1, \ldots, m-1}$. Diese eindeutige Zahl $b$ bezeichnen wir mit dem Ausdruck $a\; \% \; m$.
\end{mydefinition}

\begin{myexample}[Modulo-Operation]
Wir geben hier einige Beispiele an:
\begin{align*}
	10 \; & \% \; 7 = 3 \\
	-3 \; & \% \; 5 = 2 \\
	-17\; & \% \; 3 = 1 \\
	-7 \; & \% \; 3 = 2
\end{align*}
\end{myexample}
