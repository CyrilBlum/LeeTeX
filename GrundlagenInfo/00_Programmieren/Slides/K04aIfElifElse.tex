\begin{frame}[fragile]{Logische Ausdrücke}
    \begin{itemize}
        \item \tib{Logische} oder \tib{Boolsche Ausdrücke} sind Ausdrücke, die genau zwei verschiedene Werte annehmen können: entweder \textit{wahr} (\lstinline|True|) oder \textit{falsch} (\lstinline|False|). Beispiele:
              \begin{itemize}
                  \item \lstinline|3 == 5 # False|
                  \item \lstinline|(9 + 3) < (2 * 8) # True|
              \end{itemize}
        \item Weitere boolsche Relationen:
              \begin{table}[H]
                  \centering
                  \footnotesize
                  \begin{tabular}{c||p{3.1cm}}
                      \textbf{Python} & \textbf{Mathematische\newline Bedeutung} \\\midrule
                      \lstinline|==|  & gleich ($=$)                             \\
                      \lstinline|!=|  & ungleich ($\neq$)                        \\
                      \lstinline|<|   & kleiner ($<$)                            \\
                      \lstinline|<=|  & kleiner oder gleich ($\leq$)             \\
                      \lstinline|>|   & grösser ($>$)                            \\
                      \lstinline|>=|  & grösser oder gleich ($\geq$)
                  \end{tabular}
              \end{table}
        \item Verwendungsart
              \begin{lstlisting}
    if BOOLSCHER_AUSDRUCK:
        # dieser Code wird nicht ausgeführt, wenn BOOLSCHER_AUSDRUCK falsch (False) ist.
    \end{lstlisting}
    \end{itemize}
\end{frame}
