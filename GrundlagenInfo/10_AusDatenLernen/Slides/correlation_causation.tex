\begin{frame}{Definition: Kausalität}
    \begin{mydefinition}[Kausalität]
    \textbf{Kausalität} beschreibt:
    \begin{itemize}
        \item Ursache-Wirkung-Beziehung zwischen zwei Variablen A und B.
        \item Ursprung des Begriffs: lat. \textit{causa} = Ursache
        \item Bedarf eines Nachweises (z.B. Experimente oder statistische Analysen).
        \item Verschiedenen Arten von Kausalität existieren.
    \end{itemize}
    \end{mydefinition}
\end{frame}

\begin{frame}{Arten von Kausalität: Koinzidenz (=Zufall)}
    \textbf{Koinzidenz} beschreibt eine zufällige Korrelation zwischen zwei Variablen, die keine kausale Beziehung haben.
    \begin{center}
        \begin{tikzpicture}
            \node[circle, draw] (A) at (0,0) {A};
            \node[circle, draw] (B) at (3,0) {B};
            \draw[<->, dashed] (A) -- (B) node[midway, above] {zufällig};
        \end{tikzpicture}
    \end{center}
    \begin{center}
        Beispiel: \textit{Der Pro-Kopf-Verbrauch von Mozzarella-Käse sowie die verliehenen Bauingenieurs-Doktortitel in den USA sind zwischen den Jahren 2000 bis 2009 beide angestiegen.}
    \end{center}
\end{frame}

\begin{frame}{Arten von Kausalität: Zyklische Beeinflussung}
    \textbf{Zyklische Beeinflussung} beschreibt eine Situation, in der zwei Variablen sich gegenseitig beeinflussen. Zum Beispiel könnte A B beeinflussen und B wiederum A beeinflussen.
    \begin{center}
        \begin{tikzpicture}
            \node[circle, draw] (A) at (0,0) {A};
            \node[circle, draw] (B) at (3,0) {B};
            \draw[->] (A) to[bend left] (B);
            \draw[->] (B) to[bend left] (A);
        \end{tikzpicture}
    \end{center}
    \begin{center}
        Beispiel: \textit{Schlafmangel führt zu mehr Stress, was wiederum zu Schlafmangel führt.}
    \end{center}
\end{frame}

\begin{frame}{Arten von Kausalität: Gemeinsamer Grund}
    \textbf{Gemeinsamer Grund} beschreibt eine Situation, in der eine dritte Variable sowohl A als auch B beeinflusst. Zum Beispiel könnte C sowohl A als auch B beeinflussen.
    \begin{center}
        \begin{tikzpicture}
            \node[circle, draw] (C) at (1.5,1.5) {C};
            \node[circle, draw] (A) at (0,0) {A};
            \node[circle, draw] (B) at (3,0) {B};
            \draw[->] (C) -- (A);
            \draw[->] (C) -- (B);
        \end{tikzpicture}
    \end{center}
    \begin{center}
        Beispiel: \textit{Zwischen November und Januar stiegen sowohl der Verkauf von Winterreifen als auch die Häufigkeit von Erkältungen an (Grund: Jahreszeit).}
    \end{center}
\end{frame}

\begin{frame}{Arten von Kausalität: Indirekte Kausalität}
    \textbf{Indirekte Kausalität} beschreibt eine Situation, in der A B beeinflusst, aber nicht direkt. Zum Beispiel könnte A C beeinflussen und C B beeinflussen.
    \begin{center}
        \begin{tikzpicture}
            \node[circle, draw] (A) at (0,0) {A};
            \node[circle, draw] (C) at (1.5,1.5) {C};
            \node[circle, draw] (B) at (3,0) {B};
            \draw[->] (A) -- (C);
            \draw[->] (C) -- (B);
        \end{tikzpicture}
    \end{center}
    \begin{center}
        Beispiel: \textit{Sport führt zu besserem Schlaf, was wiederum zu besseren Prüfungsergebnissen führt.}
    \end{center}
\end{frame}

\begin{frame}{Arten von Kausalität: Direkte Kausalität}
    \textbf{Direkte Kausalität} beschreibt eine Situation, in der A B direkt beeinflusst.
    \begin{center}
        \begin{tikzpicture}
            \node[circle, draw] (A) at (0,0) {A};
            \node[circle, draw] (B) at (3,0) {B};
            \draw[->] (A) -- (B);
        \end{tikzpicture}
    \end{center}
    \begin{center}
        Beispiel: \textit{Mehr Video-Streams führen automatisch zu mehr Geld für den Youtuber.}
    \end{center}
\end{frame}

\begin{frame}[fragile]{Auftrag}{(Aufgabe 1.2 im Skript)}

            Welche Art der Beeinflussung liegt vor? 
            \textbf{(a)} Koinzidenz, 
            \textbf{(b)} Zyklische Beeinflussung, 
            \textbf{(c)} Gemeinsamer Grund, 
            \textbf{(d)} Indirekte Kausalität, 
            \textbf{(e)} Direkte Kausalität?

            \begin{enumerate}
                \item Die Durchschnittsnote in der Klasse steigt über das Schuljahr hinweg. Gleichzeitig wird mehr Kaffee im Lehrerzimmer konsumiert.
                \item In den Sommermonaten nehmen sowohl die Anzahl Sonnenbrände wie auch der Verkauf von Glacé zu.
                \item In den letzten zehn Jahren hat der Konsum von Avocados in der Schweiz zugenommen. Gleichzeitig ist die Zahl bestandener Maturaprüfungen gestiegen.
                \item Eine Messung hat ergeben, dass Menschen, die viel Sport treiben, bessere Prüfungsergebnisse erzielen.
                \item Eine Untersuchung von 1000 SchülerInnnen hat gezeigt, dass SchülerInnen, welche gestresst sind, schlechtere Noten erzielen.
            \end{enumerate}
\end{frame}

\begin{frame}{Lösungsvorschlag: Durchschnittsnote und Kaffee}
    \begin{center}
        ``Die Durchschnittsnote in der Klasse steigt über das Schuljahr hinweg. Gleichzeitig wird mehr Kaffee im Lehrerzimmer konsumiert.''
    \end{center}
    \only<2->{\textbf{Lösung:} \textbf{Koinzidenz}: Keine kausale Beeinflussung. Die Korrelation ist zufällig oder durch eine Drittvariable erklärbar (z.B. mehr Stress $\rightarrow$ mehr Kaffee).
        \begin{center}
            \begin{tikzpicture}
                \node[circle, draw] (A) at (0,0) {\Large\emoji{hundred-points}};
                \node[circle, draw] (B) at (4,0) {\Large\emoji{coffee}};
                \draw[<->, dashed] (A) -- (B) node[midway, above] {zufällig (?)};
            \end{tikzpicture}
        \end{center}}
\end{frame}

\begin{frame}{Lösungsvorschlag: Sonnenbrände und Glacé}
    \begin{center}
        ``In den Sommermonaten nehmen sowohl die Anzahl Sonnenbrände wie auch der Verkauf von Glacé zu.''
    \end{center}
    \only<2->{\textbf{Lösung:} \textbf{Gemeinsamer Grund}: Beide Grössen werden durch eine Drittvariable beeinflusst: das Wetter bzw. die Jahreszeit.
        \begin{center}
            \begin{tikzpicture}
                \node[circle, draw] (C) at (1.5,1.5) {\Large\emoji{sun}};
                \node[circle, draw] (A) at (0,0) {\Large\emoji{ice-cream}};

            \end{tikzpicture}
        \end{center}}
\end{frame}

\begin{frame}{Definition: Korrelation}
    \begin{mydefinition}[Korrelation]
    \textbf{Korrelation} beschreibt:
    \begin{itemize}[<+->]
        \item Statistische Beziehung zwischen zwei Variablen.
        \item Ursprung des Begriffs:
              \begin{itemize}
                  \item \textit{co(n)-} = \enquote{mit}.
                  \item \textit{relatio} = Beziehung.
              \end{itemize}
        \item Bedeutung: Gegenseitige Beziehung oder Wechselbeziehung (\emoji{arrows-counterclockwise}).
        \item Häufig als Zahl ausgedrückt (zwischen -1 und 1).
    \end{itemize}
    \end{mydefinition}
\end{frame}
