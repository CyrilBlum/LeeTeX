\begin{frame}{Was sind Markov-Ketten?}
    \begin{tcolorbox}[colback=blue!5!white, colframe=blue!75!black, boxrule=0.8pt, enhanced, left=2mm, right=2mm, top=1mm, bottom=1mm, box align=base, title=Andrej A. Markov]
        \begin{minipage}{0.25\textwidth}
            \includegraphics[width=\linewidth]{GrundlagenInfo/10_AusDatenLernen/Figures/AAMarkov.png}
        \end{minipage}
        \hfill
        \begin{minipage}{0.7\textwidth}
            \begin{itemize}[<+->]
                \item Die Markov-Kette ist benannt nach dem russischen Mathematiker \textbf{Andrej Andrejewitsch Markov} (1856–1922).
                \item Markov entdeckte die Markov-Ketten bei der Analyse von Buchstabenfolgen in literarischen Texten (Textanalyse).
            \end{itemize}
        \end{minipage}
    \end{tcolorbox}
    \only<3->{
        \begin{mydefinition}
    \begin{itemize}[<+->]
        \item Mathematisches Modell für \textbf{Abfolgen von Zuständen}
        \item \textbf{Markov-Eigenschaft}: Der nächste Zustand hängt \textbf{nur vom vorherigen Zustand} ab
    \end{itemize}
    \end{mydefinition}
    }
\end{frame}

\begin{frame}
    \begin{myexercise}[Markov-Eigenschaft erfüllt?]
        \begin{enumerate}
            \item Das Wetter morgen kann gut vorhergesagt werden basierend auf dem heutigen Wetter.
            \item Die Wahrscheinlichkeit, dass ein Schüler heute seine Hausaufgaben macht, hängt davon ab, ob er sie gestern und vorgestern gemacht hat.
            \item Die Wahrscheinlichkeit, dass ein Programm abstürzt, hängt davon ab, wie viele andere Programme bereits abgestürzt sind.
            \item Die Wahrscheinlichkeit, dass eine Person Kleider einer bestimmten Marke kauft, hängt nur davon ab, ob das zuletzt gekaufte Kleidungsstück von dieser Marke war.
        \end{enumerate}
    \end{myexercise}
\end{frame}

\begin{frame}{Startvektor und Vorhersage zukünftiger Zustände}
    \begin{itemize}[<+->]
        \item Startvektor $\pi_0$ gibt den Anfangszustand (=Wahrscheinlichkeit, in einem Zustand zu sein) an
        \item Beispiel: Heute ist es regnerisch
              \[
                  \begin{aligned}
                      \pi_0 =
                       & \begin{pmatrix}
                             0 & 1
                         \end{pmatrix}                                                     \\
                       & \begin{matrix}
                             \;\; \Uparrow & \Uparrow
                         \end{matrix}                                           \\
                       & \begin{matrix}
                             \; \text{\emoji{sun}} & \text{\emoji{umbrella}}
                         \end{matrix}
                  \end{aligned}
              \]
              Wie hoch ist die Wahrscheinlichkeit, dass es morgen sonnig wird?
        \item Die Entwicklung nach einem Schritt: $\pi_1 = \pi_0 \cdot P$
    \end{itemize}

    \only<4->{
        \begin{align*}
            \pi_1 = \pi_0 \cdot P & =
            \begin{pmatrix}
                0 & 1 \\
            \end{pmatrix}
            \cdot
            \begin{pmatrix}
                0.8 & 0.2 \\
                0.4 & 0.6 \\
            \end{pmatrix}            \\
                                  & =
            \begin{pmatrix}
                0.4 & 0.6 \\
            \end{pmatrix}
        \end{align*}
    }
\end{frame}

\begin{frame}{Matrixmultiplikation}
    \only<1-3>{\begin{itemize}[<+->]
            \item Um die Entwicklung einer Markov-Kette zu berechnen, multiplizieren wir einen Zeilenvektor mit einer Matrix.
        \end{itemize}}

    \only<1>{
        \[
            \begin{array}{c}
                \Big(
                \fcolorbox{red!10}{red!10}{$a \;\; b$}
                \Big)
                \cdot
                \left(
                \begin{array}{>{\columncolor{red!10}}c c}
                        e & f \\
                        g & h \\
                    \end{array}
                \right)
                =
                \Big(
                \fcolorbox{red!10}{red!10}{$a \cdot e + b \cdot g$}\fcolorbox{white}{white}{$\;\; a \cdot f + b \cdot h$}
                \Big)
            \end{array}
        \]}

    \only<2>{
        \[
            \begin{array}{c}
                \Big(
                \fcolorbox{red!10}{red!10}{$a \;\; b$}
                \Big)
                \cdot
                \left(
                \begin{array}{c >{\columncolor{red!10}}c}
                        e & f \\
                        g & h \\
                    \end{array}
                \right)
                =
                \Big(
                \fcolorbox{white}{white}{$a \cdot e + b \cdot g$}\fcolorbox{red!10}{red!10}{$\;\; a \cdot f + b \cdot h$}
                \Big)
            \end{array}
        \]}
    \only<3->{
        \[
            \begin{pmatrix}
                \textcolor{red}{a} & \textcolor{blue}{b} \\
            \end{pmatrix}
            \cdot
            \begin{pmatrix}
                \textcolor{teal}{e}   & \textcolor{violet}{f} \\
                \textcolor{orange}{g} & \textcolor{brown}{h}  \\
            \end{pmatrix}
            =
            \begin{pmatrix}
                \textcolor{red}{a} \cdot \textcolor{teal}{e} + \textcolor{blue}{b} \cdot \textcolor{orange}{g} & \textcolor{red}{a} \cdot \textcolor{violet}{f} + \textcolor{blue}{b} \cdot \textcolor{brown}{h} \\
            \end{pmatrix}
        \]
    }
    \only<4->{
        \begin{myexercise}[\faCalculator{} Matrix-Multiplikation]
            Multiplizieren Sie $\begin{pmatrix} \textcolor{red}{0.7} & \textcolor{blue}{0.3} \end{pmatrix} \cdot \begin{pmatrix} \textcolor{teal}{0.8} & \textcolor{violet}{0.2} \\ \textcolor{orange}{0.4} & \textcolor{brown}{0.6} \end{pmatrix}$.
            \only<5->{
                \begin{myanswer}
                    \[
                        \begin{aligned}
                             &
                            \begin{pmatrix}
                                \textcolor{red}{0.7} & \textcolor{blue}{0.3} \\
                            \end{pmatrix}
                            \cdot
                            \begin{pmatrix}
                                \textcolor{teal}{0.8}   & \textcolor{violet}{0.2} \\
                                \textcolor{orange}{0.4} & \textcolor{brown}{0.6}  \\
                            \end{pmatrix}
                            \\&=
                            \begin{pmatrix}
                                \textcolor{red}{0.7} \cdot \textcolor{teal}{0.8} + \textcolor{blue}{0.3} \cdot \textcolor{orange}{0.4} & \textcolor{red}{0.7} \cdot \textcolor{violet}{0.2} + \textcolor{blue}{0.3} \cdot \textcolor{brown}{0.6} \\
                            \end{pmatrix}
                            \\&=
                            \begin{pmatrix}
                                0.56 + 0.12 & 0.14 + 0.18 \\
                            \end{pmatrix}
                            =
                            \begin{pmatrix}
                                0.68 & 0.32 \\
                            \end{pmatrix}
                        \end{aligned}\]
                \end{myanswer}
            }
        \end{myexercise}
    }
\end{frame}

\begin{frame}{Wettervorhersage für übermorgen}
    \begin{itemize}[<+->]
        \item Startvektor: \(\pi_0 = \begin{pmatrix} 0 & 1 \end{pmatrix}\) \hspace{0.5em} (heute regnerisch)
        \item Nach einem Tag:
              \[
                  \pi_1 = \pi_0 \cdot P = \begin{pmatrix} 0 & 1 \end{pmatrix} \cdot \begin{pmatrix} 0.8 & 0.2 \\ 0.4 & 0.6 \end{pmatrix} = \begin{pmatrix} 0.4 & 0.6 \end{pmatrix}
              \]
        \item Nach zwei Tagen:
              \[
