\begin{frame}{Fehler in binärer Kodierung}
    \begin{itemize}
        \item In Computersystemen werden Informationen durch Bitfolgen kodiert, z.\,B.\ A~$\rightarrow$~01000001
        \item Ein häufiger Fehler beim Übertragen von Informationen ist das Umflippen eines Bits (0~$\rightarrow$~1 bzw. 1~$\rightarrow$~0)
        \item Eine Kodierung heißt \textbf{k-fehlererkennend}, wenn das Umflippen von 1, 2, \dots, bis $k$ Bits in der Nachricht als Fehler erkannt wird
    \end{itemize}
\end{frame}

\begin{frame}{Qualität einer Kodierung}
    \textbf{Qualität}: Wie viele Fehler können garantiert erkannt werden?
    \begin{table}[H]
        \centering
        \begin{tabular}{c|c|c|c}
                             & \textbf{Kodierung$_1$} & \textbf{Kodierung$_2$}     & \textbf{Kodierung$_3$}                \\
            \midrule\midrule
            \textbf{Weiss}   & 00                     & 00\textcolor{orange}{0000} & 00\textcolor{orange}{00\underline{0}} \\
            \midrule
            \textbf{Gelb}    & 01                     & 01\textcolor{orange}{0101} & 01\textcolor{orange}{01\underline{1}} \\
            \midrule
            \textbf{Blau}    & 10                     & 10\textcolor{orange}{1010} & 10\textcolor{orange}{10\underline{1}} \\
            \midrule
            \textbf{Schwarz} & 11                     & 11\textcolor{orange}{1111} & 11\textcolor{orange}{11\underline{0}} \\
        \end{tabular}
    \end{table}
    Welche Kodierung ist am besten?
    \begin{itemize}
        \item Kodierung$_1$: 0 Fehler erkannt
        \item Kodierung$_2$: 2 Fehler erkannt
        \item Kodierung$_3$: 2 Fehler erkannt (aber kürzer!)
    \end{itemize}
\end{frame}

\begin{frame}{Abstand einer Kodierung}{ANNA, ENZO und HANS}
    \begin{table}[H]
        \centering
        \begin{tabular}{c@{\hspace{1.5cm}}c@{\hspace{1.5cm}}c}
            \begin{tabular}{c}
                A N N A             \\
                \underline{E N Z O} \\
                x o x x             \\
                \textcolor{orange}{Abstand 3}
            \end{tabular}
             &
            \begin{tabular}{c}
                A N N A             \\
                \underline{H A N S} \\
                x x o x             \\
                \textcolor{orange}{Abstand 3}
            \end{tabular}
             &
            \begin{tabular}{c}
                E N Z O             \\
                \underline{H A N S} \\
                x x x x             \\
                \textcolor{orange}{Abstand 4}
            \end{tabular}
        \end{tabular}
        \caption{Abstände zwischen den Codewörtern ANNA, ENZO und HANS}
    \end{table}
    \begin{itemize}
        \item Abstand dieser Kodierung: \textcolor{red}{3}
        \item Abstand = Minimum aller paarweisen Abstände der Codewörter
        \item Bis zu zwei Buchstaben können geändert werden, ohne dass man von einem Wort zu einem anderen übergeht
        \item Beispiel: \enquote{ENZO} mutiert zu \enquote{\red{A}N\red{N}O} $\rightarrow$ kein anderes gültiges Wort, also Fehler erkannt
    \end{itemize}
\end{frame}
