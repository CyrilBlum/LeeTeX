\begin{frame}{Was kann eine 0 / 1 bedeuten?}
	\begin{itemize}
		\item Zahl 1 / 0
		\item wahr / falsch (Logik)
		\item An / Aus (z.B. Strom oder Licht)
		\item schwarz / weiss (z.B. Barcode)
		\item positiv / negativ gepolt (z.B. Magnet auf Festplatte)
		\item leitet / leitet nicht (z.B. Memory Stick)
		\item reflektiert / reflektiert nicht (z.B. CD oder DVD)
		\item ``to be or not to be''
	\end{itemize}
	$\rightarrow$ ProgrammiererIn entscheidet!
\end{frame}

\begin{frame}[fragile]{Rechnen mit Zahlen}
	Nur 5 Regeln!
	\begin{itemize}
	\item 0 + 0 = 0 
	\item 0 + 1 = 1 
	\item 1 + 0 = 1 
	\item 1 + 1 = 0, Übertrag 1
	\item 1 + 1 + 1 = 1, Übertrag 1
	\end{itemize}
	
	Addideren in Binärsystem ($101011_2$ + $111010_2$):
	\[
	\begin{array}{ccccccc}
		& 1 & 0 & 1 & 0 & 1 & 1_2 \\
		+ & \underset{1}{1} & \underset{1}{1} & \underset{}{1} & \underset{1}{0} & \underset{}{1} & \underset{}{0_2} \\
		\hline
		1 & 1 & 0 & 0 & 1 & 0 & 1_2 \\\hline\hline
	\end{array}
	\]
\end{frame}
