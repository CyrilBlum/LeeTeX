\begin{frame}{Bits \& Bytes}
	\begin{itemize}[<+->]
		\item \textbf{bit} = ``\textit{\underline{bi}nary digi\underline{t}}'', ``binäre Ziffer'' (eine 0 oder eine 1)
		\item \textbf{byte} = ``Bissen'' (von engl. \textit{``bite''}=Bissen aus 8 bits) \emoji{yum}
	\end{itemize}
\end{frame}

\begin{frame}{Was kann eine 0 / 1 bedeuten?}
	\begin{itemize}[<+->]
		\item Zahl 1 / 0
		\item wahr / falsch (Logik)
		\item An / Aus (z.B. Strom oder Licht)
		\item schwarz / weiss (z.B. Barcode)
		\item positiv / negativ gepolt (z.B. Magnet auf Festplatte)
		\item verbunden / getrennt (z.B. Memory Stick)
		\item ``to be or not to be''\\
		$\rightarrow$ ProgrammiererIn entscheidet!
	\end{itemize}
	
\end{frame}

\begin{frame}[fragile]{Rechnen mit Zahlen}

	\begin{array}{ccccccc}
	  & 0 & 4 & 9 & 0 & 3 & 5_{10} \\
	+ & \underset{}{7} & \underset{1}{1} & \underset{}{8} & \underset{1}{2} & \underset{}{9} & \underset{}{3_{10}} \\
	\hline
	 & 7 & 6 & 7 & 3 & 2 & 8_{10} \\\hline\hline
	\end{array}

\end{frame}

\begin{frame}[fragile]{Rechnen mit Zahlen}
	Nur 5 Regeln!
	\begin{itemize}
	\item 0 + 0 = 0 
	\item 0 + 1 = 1 
	\item 1 + 0 = 1 
	\item 1 + 1 = 0, Übertrag 1
	\item 1 + 1 + 1 = 1, Übertrag 1
	\end{itemize}
	
	Addideren in Binärsystem ($101011_2$ + $111010_2$):
	\[
	\begin{array}{ccccccc}
		& 1 & 0 & 1 & 0 & 1 & 1_2 \\
		+ & \underset{1}{1} & \underset{1}{1} & \underset{}{1} & \underset{1}{0} & \underset{}{1} & \underset{}{0_2} \\
		\hline
		1 & 1 & 0 & 0 & 1 & 0 & 1_2 \\\hline\hline
	\end{array}
	\]
\end{frame}
