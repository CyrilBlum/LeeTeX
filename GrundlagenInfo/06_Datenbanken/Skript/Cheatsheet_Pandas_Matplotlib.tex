% Pandas Basics
\begin{cheatSheetBox}[title=Importieren]
\begin{lstlisting}[language=Python]
import pandas as pd
import matplotlib.pyplot as plt
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=CSV-Datei laden]
\begin{lstlisting}[language=Python]
df = pd.read_csv('datei.csv')
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Erste/letzte Zeilen anzeigen]
\begin{lstlisting}[language=Python]
df.head()   # erste 5 Zeilen
df.tail()   # letzte 5 Zeilen
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Spalten auswählen]
\begin{lstlisting}[language=Python]
df['Spalte']           # einzelne Spalte
df[['A', 'B']]         # mehrere Spalten
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Zeilen filtern]
\begin{lstlisting}[language=Python]
df[df['Alter'] > 18]   # Filter
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Statistiken]
\begin{lstlisting}[language=Python]
df.describe()          # Übersicht
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Gruppieren]
\begin{lstlisting}[language=Python]
df.groupby('Spalte').mean()
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Neue Spalte erstellen]
\begin{lstlisting}[language=Python]
df['BMI'] = df['Gewicht'] / (df['Größe']/100)**2
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Fehlende Werte]
\begin{lstlisting}[language=Python]
df.isnull().sum()      # Anzahl fehlender Werte
df.dropna()            # Zeilen mit NaN entfernen
df.fillna(0)           # NaN ersetzen
\end{lstlisting}
\end{cheatSheetBox}

% Matplotlib Basics
\begin{cheatSheetBox}[title=Plotten mit Matplotlib]
\begin{lstlisting}[language=Python]
plt.plot(df['x'], df['y'])
plt.xlabel('x')
plt.ylabel('y')
plt.title('Mein Plot')
plt.show()
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Schnelles Plotten mit Pandas]
\begin{lstlisting}[language=Python]
df.plot(x='x', y='y', kind='line')
plt.show()
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Weitere Plottypen]
\begin{lstlisting}[language=Python]
df.plot(kind='bar')      # Balkendiagramm
df.plot(kind='hist')     # Histogramm
df.plot(kind='box')      # Boxplot
df.plot(kind='scatter', x='A', y='B') # Streudiagramm
\end{lstlisting}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Speichern als Bild]
\begin{lstlisting}[language=Python]
plt.savefig('plot.png')
\end{lstlisting}
\end{cheatSheetBox}
