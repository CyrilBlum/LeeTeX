\begin{frame}[fragile]{Schichten: Übersicht}
\begin{table}[h]
\centering\tiny
\begin{tabular}{c|p{2cm}|p{4cm}|p{1.5cm}|p{1.5cm}}
\textbf{\#} & \textbf{Schichtname}           & \textbf{Verantwortlichkeit}                                                                                & \textbf{Informationsform (Dateneinheit)} & \textbf{Gerät oder Protokoll}               \\ \midrule\midrule
7                    & Anwendungsschicht              & Hilft bei der Identifizierung des Clients und synchronisiert die Kommunikation.                            & Nachricht                               & SMTP                                        \\ \midrule
6                    & Präsentationsschicht           & Daten von der Anwendungsschicht werden extrahiert und für die Übertragung in das erforderliche Format gebracht. & Nachricht                               & JPEG, MPEG, GIF                            \\ \midrule
5                    & Sitzungsschicht                & Stellt Verbindung her, hält diese aufrecht, gewährleistet Authentifizierung und sichert die Sicherheit.   & Nachricht (oder verschlüsselte Nachricht) & Gateway, NetBIOS, PPTP                                    \\ \midrule
4                    & Transportschicht               & Nimmt Dienst von der Netzwerkschicht und stellt ihn der Anwendungsschicht zur Verfügung.                  & Segment                                 & Firewall, TCP, Ports                                   \\ \midrule
3                    & Netzwerkschicht                & Übertragung von Daten von einem Host zu einem anderen, der sich in unterschiedlichen Netzwerken befindet. & Paket                                   & Router, IPv4, IPv6                                     \\ \midrule
2                    & Verbindungsschicht             & Übermittlung von Nachrichten von Knoten zu Knoten.                                                        & Frame                                  & Switch, Bridge, MAC                            \\ \midrule
1                    & Physische Schicht              & Herstellung physischer Verbindungen zwischen Geräten.                                                     & Bits                                    & Hub, Repeater, Modem, Kabel                \\
\end{tabular}
\caption*{Übersicht der Netzwerkschichten und ihrer Funktionen}
\end{table}
