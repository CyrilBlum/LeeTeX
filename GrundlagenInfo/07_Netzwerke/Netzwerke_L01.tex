\begin{frame}[fragile]{Schichten \& Protokolle}
\begin{columns}[t]
\begin{column}{.5\textwidth}
\textbf{Schichten}
\begin{itemize}[<+->]
\item Z.B. Anwendungs-, Transport-, Vermittlungs-, Netzzugangsschicht
\item Erlauben es, spezialisierte Aufgaben in einzelne Schichten aufzuteilen: Z.B. Übermittlung von Signalen, Sprache-Regeln etc.
\end{itemize}
\end{column}

\begin{column}{.5\textwidth}
\textbf{Protokolle}
\begin{itemize}[<+->]
\item Z.B. TCP, IP, etc.
\item Definiert Ablauf einer Kommunikation
\item Definiert den Aufbau der Datenpakete
\item Wird von Computern / Prozessen verwendet
\item Gibt es auf jeder Ebene des Schichtenmodells
\item Im echten Leben: Begrüssung, \textit{small talk}, etc.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}[fragile]{Schichtenmodell}{Vorteile}
\begin{itemize}[<+->]
\item \textbf{Modularität}: Aufbau auf besetehenden Schichten
\item \textbf{Austauschbarkeit} / Flexibilität
\begin{itemize}
\item Übertragung über Kabel, WLAN etc. kann ausgetauscht werden, ohne dass darüberliegende Schichten tangiert werden
\item Analogie: Anstelle von Lastwagen kann Motorrad verwendet werden
\end{itemize}
\end{itemize}

\end{frame}

\begin{frame}[fragile]{\faLaptop{} Client vs Server vs Host}{Anwendungsschicht}
Definitionen
\begin{itemize}[<+->]
\item \textbf{Client}
\begin{itemize}
\item ``Kunde'' (von en. \textit{client})
\item Computer-Programm, das etwas will
\end{itemize}
\item \textbf{Server}
\begin{itemize}
\item ``Diener'' (von en. \textit{to serve} = bedienen, $\neq$ ``surf'')
\item Computer-Programm, das etwas hat, das andere Computer brauchen
\item Kann Verbindungsanfragen von anderen Computern akzeptieren / ablehnen
\end{itemize}
\item \textbf{Host} 
\begin{itemize}
\item = Computer in einem Netzwerk, allgemein für ``server'' oder ``client''
\item \textit{Host name} = Name eines Computers
\end{itemize}

\end{itemize}
\end{frame}

\begin{frame}[fragile]{\faMapMarker{} IP-Adresse}{Vermittlungsschicht}

\begin{itemize}[<+->]
\item Netzwerk-Adresse eines Computers
\item Analogie: Wohnadresse einer Person (Strasse \& Stadt)
\item Häufig im IPv4-Format: 4 bytes, also 4 Zahlen von 0-255
\item Wird meist dezimal angegeben, z.B. 192.168.0.1
\item Übung: eigene IPv4-Adresse herausfinden
\begin{itemize}
\item MacOS, Terminal (via Spotlight-Suche): \\
\texttt{ipconfig getifaddr en0}
\item Windows, Programm ``cmd'': \\\texttt{ipconfig}
\end{itemize}
\item Wie viele IPv4-Adressen gibt es?
\begin{itemize}
\item $4\cdot 8 \text{ bytes} = 32 \text{ bytes}$
\item $2^{32} \sim 4 \text{ Milliarden Adressen}$
\item $\rightarrow \text{zu wenig!}$
\end{itemize}
\item IPv6: 128 bytes, fast unendlich viele Adressen
\[2^{128} \sim 3.4\cdot 10^{38}\]
\end{itemize}
\end{frame}

\begin{frame}[fragile]{\faGlobe{} Netzmasken}{Vermittlungsschicht}
\includegraphics[width=0.3\linewidth]{GrundlagenInfo/07_Netzwerke/Figures/Subnetz-2}
\textbf{Netzmaske}
\begin{itemize}[<+->]
\item Sieht ähnlich aus wie eine IP-Adresse: \commandBluebox|xxx.xxx.xxx.xxx|
\item Fasst mehrere IPs zu einer Gruppe (= Subnetz) zusammen
\item Gibt an, wie viele Stellen einer IP-Adresse innerhalb eines Subnetzes gleich sein müssen
\\
\commandbox|159.233.1.22|\\
\commandBluebox|255.255.255.0|\\
Bedeutet, dass alle IP-Adressen die mit \commandbox|159.233.1.[...]| anfangen, dazugehören
\end{itemize}
\end{frame}

\begin{frame}[fragile]{\faCompass{} \texttt{ping}}{Vermittlungsschicht}

\begin{itemize}[<+->]
\item Im Terminal / cmd: \commandbox|ping [ip address]|, z.B.:\\
\commandbox|ping 159.233.1.22|
\item Minimale Nachricht an andere IP schicken
\item Ziel: schauen ob eine IP existiert, bzw. erreichbar ist
\end{itemize}
\end{frame}
