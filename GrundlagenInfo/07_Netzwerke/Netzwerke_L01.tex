\begin{frame}[fragile]{Schichtenmodell}{Vorteile}
\begin{itemize}[<+->]
\item \textbf{Modularität}: Aufbau auf besetehenden Schichten

\begin{itemize}
\item Übertragung über Kabel, WLAN etc. kann ausgetauscht werden, ohne dass darüberliegende Schichten tangiert werden
\item Analogie: Anstelle von Lastwagen kann Motorrad verwendet werden
\end{itemize}

\begin{frame}[fragile]{Anwendungsschicht}{Client vs Server \emoji{computer}}
Beides sind Computer!
\begin{itemize}[<+->]
\item \textbf{Client}
\begin{itemize}
\item Kunde (von en. \textit{client})
\item Computer, der etwas will
\end{itemize}
\item \textbf{Server}
\begin{itemize}
\item Diener (von en. \textit{to serve} = bedienen, $\neq$ ``surf'')
\item Computer, der etwas hat, das andere Computer brauchen
\item Kann Verbindungsanfragen von anderen Computern akzeptieren / ablehnen
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Präsentationsschicht}
Beispiel: HTML (\textit{Hypertext Markup Language})
\begin{itemize}[<+->]
\item Demo (\textit{Inspect Element} oder \textit{View Page Source})
\item Speichert die Struktur einer Webseite
\item Kann weitere Dateien enthalten, beispielsweise:
\begin{itemize}
\item CSS (Cascading Style Sheet), für die Darstellung (Farbe, Grösse etc.)
\item JavaScript, jQuery etc. für die Interaktivität (Ein- und Ausblenden von Objekten etc.)
\end{itemize}
\item Wie man eine Webseite baut ist hier nicht das Hauptthema, wird aber in der Interessenwoche vermittelt
\end{itemize}

\begin{frame}[fragile]{Netzwerkschicht}{IP-Adresse}

\begin{itemize}[<+->]
\item Netzwerk-Adresse eines Computers
\item Analogie: Wohnadresse einer Person (Strasse \& Stadt)
\item Häufig im IPv4-Format: 4 bytes, also 4 Zahlen von 0-255
\item Wird meist dezimal angegeben, z.B. 192.168.0.1
\item Übung: eigene IPv4-Adresse herausfinden
\begin{itemize}
\item MacOS, Terminal (via Spotlight-Suche): \\
\texttt{ipconfig getifaddr en0}
\item Windows, Programm ``cmd'': \\\texttt{ipconfig}
\end{itemize}
\item Wie viele IPv4-Adressen gibt es?
\begin{itemize}
\item $4\cdot 8 \text{ bytes} = 32 \text{ bytes}$
\item $2^{32} \sim 4 \text{ Milliarden Adressen}$
\item $\rightarrow \text{zu wenig!}$
\end{itemize}
\item IPv6: 128 bytes, fast unendlich viele Adressen
\[2^{128} \sim 3.4\cdot 10^{38}\]
\end{itemize}

\end{frame}

\begin{frame}[fragile]{Verbindungsschicht}{MAC-Adresse}
\begin{itemize}[<+->]
\item So etwas wie der ``Fingerabdruck'' eines Computers
\item Analogie: Übergabe Brief von einer Hand in die andere
\item 48 bit, die normalerweise in hexadezimaler Weise angegeben werden
\item Z.B. \texttt{48-2C-6A-1E-59-3D}
\item Jede Netzwerkkarte hat eine MAC-Adresse
\end{itemize}
\end{frame}
