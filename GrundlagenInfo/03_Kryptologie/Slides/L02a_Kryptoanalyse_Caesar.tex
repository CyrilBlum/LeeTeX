	Wie würden Sie folgenden Text knacken?\\[.5cm]
\small\begin{verbatim}
NGKALOMYZKXHAINYZGHKOTJKAZYINKTZKDZKTOYZSKOYZKTYK...
\end{verbatim}
	    
	 \begin{itemize}[<+->]
	   \item Häufigster Buchstabe \textcolor{ForestGreen}{im deutschen Alphabet}: \texttt{E} (= Buchstabe 5)
	   \item Häufigster Buchstabe im obigen \textcolor{RoyalBlue}{im obigen Geheimtext}: \texttt{K} (= Buchstabe 11)
	   \item Verschiebung von $11-5 = 6$ 
	   \item Schlüssel = 6 (= Buchstabe \texttt{G})
	 \end{itemize}
