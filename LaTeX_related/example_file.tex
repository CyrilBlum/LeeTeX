\chapter{Beispiele verschiedener Umgebungen}

\begin{aufgabe}[Polynomfunktion auswerten]{aufgabe:evalPoly}
Gegeben sei die Polynomfunktion
\begin{align*}
  p:  \R &\to \R, \\
      x  &\mapsto x^2.
  \end{align*}
Bestimmen Sie die Zahl $p(3)$.
\end{aufgabe}

\begin{antwort}{aufgabe:evalPoly}
Die gesuchte Zahl ist gegeben durch
\begin{align*}
  p(3) = 3^2 = 9.    
\end{align*}
\end{antwort}
