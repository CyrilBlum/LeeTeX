Folgende Boxen zeigen Beispiele, wie die in diesem Projekt definierten \textit{custom environments} verwendet werden können. Links steht jeweils der LaTeX-code und rechts das kompilierte Resultat.

\begin{tcblisting}{colback=red!5!white,
colframe=red!75!black,title=\texttt{aufgabe}-Umgebung mit \texttt{antwort}-Umgebung, 
fonttitle=\bfseries, 
listing side text}
\begin{aufgabe}[Polynomfunktion auswerten]{aufgabe:evalPoly}
Gegeben sei die Polynomfunktion
\begin{align*}
  p:  \R &\to \R, \\
      x  &\mapsto x^2.
  \end{align*}
Bestimmen Sie die Zahl $p(3)$.
\end{aufgabe}

\begin{antwort}{aufgabe:evalPoly}
Die gesuchte Zahl ist gegeben durch
\begin{align*}
  p(3) = 3^2 = 9.    
\end{align*}
\end{antwort}

\shipoutAnswer
\end{tcblisting}
