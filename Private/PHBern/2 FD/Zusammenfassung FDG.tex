\section{Anforderungen für die Prüfung}
\begin{itemize}
  \item Kenntnis von Lehrplänen (kantonal/ausserkantonal)
  \item Didaktische und methodische Zugänge zu sozial- und naturwissenschaftlichen Themen
  \item Einbezug von Präkonzepten und Alltagserfahrungen der Lernenden
  \item Analyse und didaktische Rekonstruktion komplexer Inhalte
  \item Grundfertigkeiten: Beobachten, beschreiben, analysieren, vernetzen, experimentieren etc.
  \item Unterrichtskritik und Weiterentwicklung
  \item Ziel- und Kompetenzorientierung im eigenen Unterricht
  \item Durchführung von Unterricht an verschiedenen Lernorten
\end{itemize}

\section{Fokussierende Lernaufgaben (AEL)}
\textbf{Definition:} Aufgaben mit explizitem Lernfokus, die gezielt zentrale Kompetenzen fördern.

\subsection{Merkmale}
\begin{itemize}
  \item Klarer Kompetenz- oder Lernzielbezug
  \item Fordern eigenständiges Denken, Erklären, Begründen, Anwenden
  \item Unterschiedliche Lösungswege, Transferförderung
  \item Problemorientiert, lebensweltbezogen, herausfordernd
  \item Unterstützen Conceptual Change
  \item Differenzierung möglich (Basis- und Erweiterungsaufgaben)
\end{itemize}

\subsection{Didaktische Funktion}
\begin{itemize}
  \item Aktivieren Vorwissen, regen zum Nachdenken an
  \item Fördern nachhaltiges, kompetenzorientiertes Lernen
  \item Ermöglichen formative Rückmeldung und Selbstüberprüfung
  \item Unterstützen Problemlöse- und Urteilskompetenz
\end{itemize}

\subsection{Beispiele}
\begin{itemize}
  \item Erkläre, warum es in Mali zu Landnutzungskonflikten kommt.
  \item Entwickle ein Modell zur Entstehung von Stadtklima.
  \item Analysiere eine aktuelle Karte und leite Folgen für die Bevölkerung ab.
  \item Entwickle eine eigene Fragestellung zu einem aktuellen geografischen Problem.
\end{itemize}

\vspace{1em}
\textit{Hinweis:} AEL sind zentrale Bausteine für kompetenzorientierten Geografieunterricht. Sie steuern Lernprozesse, machen Lernstände sichtbar und fördern aktive Lernbeteiligung.
