\begin{frame}{Darstellung von Informationen}
    \begin{itemize}[<+->]
        \item Laute $\rightarrow$ A, B, C, \ldots{} (26 Zeichen)
        \item A $\rightarrow$ 01000001, B $\rightarrow$ 01000010, C $\rightarrow$ 01000011
              \begin{itemize}
                  \item ASCII
                  \item Vorteil: Benötigt ein Alphabet aus nur zwei Zeichen
              \end{itemize}
        \item A $\rightarrow$ 01, B $\rightarrow$ 101, C $\rightarrow$ 011 \ldots
              \begin{itemize}
                  \item Huffman \& Co
                  \item Vorteil: braucht je nach Text weniger Ressourcen (Speicher)
              \end{itemize}
        \item A $\rightarrow$ 01000001\red{0}, B $\rightarrow$ 010000010\red{0}, C $\rightarrow$ 010000011\red{1}
              \begin{itemize}
                  \item Vorteil: Fehler werden erkannt
              \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{...kann in Realität nicht passieren}
    \begin{itemize}[<+->]
        \item IBAN enthalten Prüfziffern: \texttt{CH93 0076 2011 6238 5295 7}
        \item Berechnen sich aus den restlichen Ziffern mittels festgelegter Formel
        \item Ändern einer Ziffer $\rightarrow$ IBAN ungültig, da Prüfziffer nicht mehr stimmt
        \item Fehler wird \enquote{erkannt}
    \end{itemize}
\end{frame}

\begin{frame}{Prüfziffer im EAN}
    EAN besteht aus 12 Ziffern + \red{eine Prüfziffer}
    \begin{equation}\label{eq:ean}
        x_1x_2x_3x_4x_5x_6x_7x_8x_9x_{10}x_{11}x_{12}\textcolor{red}{x_{13}}
    \end{equation}
    \only<2->{
        Die Prüfziffer wird folgendermassen berechnet:
        \begin{enumerate}
            \item $\textcolor{ForestGreen}{s}=x_1+x_3+ x_5+ x_7+ x_9+ x_{11}+3\times( x_2+ x_4+ x_6+ x_8+ x_{10}+ x_{12})$
            \item Anschliessend wird $\textcolor{red}{x_{13}}$ so gewählt, dass $\textcolor{ForestGreen}{s}+\textcolor{red}{x_{13}}$ durch 10 teilbar ist.
        \end{enumerate}
    }
\end{frame}
