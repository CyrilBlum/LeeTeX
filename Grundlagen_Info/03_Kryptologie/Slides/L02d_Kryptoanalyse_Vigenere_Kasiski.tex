  \only<2-8,11->{\texttt{C\only<1-4>{ODE}\only<5->{\gold{ODE}}CO\only<1-4>{DEC}\only<5->{\red{DEC}}ODECODECO\only<1-4>{DEC}\only<5->{\red{DEC}}OD\only<1-4>{ECO}\only<5->{\green{ECO}}DECOD\only<1-4>{ECO}\only<5->{\green{ECO}}}}

  \only<9>{\texttt{ABCDEFGHIJKLABCDEFGHIJKLABCDEFGHIJ??}}
  \only<10>{\sout{\texttt{ABCDEFGHIJKLABCDEFGHIJKLABCDEFGHIJ??}}}

  \only<3->{\texttt{O\only<1-3>{SLR}\only<4->{\annotKasiski{Gold}{S}{LR}{2}}MZ\only<1-3>{HMP}\only<4->{\annotKasiski{red}{H}{MP}{7}}SUVGWPWEV\only<1-3>{HMP}\only<4->{\annotKasiski{red}{H}{MP}{19}}HN\only<1-3>{IKB}\only<4->{\annotKasiski{ForestGreen}{I}{KB}{24}}HVBIV\only<1-3>{IKB}\only<4->{\annotKasiski{ForestGreen}{I}{KB}{32}\\[.5cm]}}}

  \begin{itemize}
    \item<4-> \enquote{\red{HMP}} und \enquote{\green{IKB}} kommen je zweimal im Geheimtext vor

          \begin{itemize}
            \item<6-> \texttt{HMP}: Positionen 7 und 19. $\rightarrow$ 19-7 = 12 = $\blue{\mathbf{2\times 2}} \times 3$
                  \only<12->{
                    \begin{tcolorbox}
                      $\mathbf{\underset{\text{Position 1 \texttt{HMP}}}{7}+\underset{\text{Codewort-Länge!}}{\blue{4}}\times \underset{\text{Anzahl Codewörter}}{3} = \underset{\text{Position 2 \texttt{HMP}}}{19}}$
                    \end{tcolorbox}
                  }
            \item<7-> \texttt{IKB}: Positionen 24 und 32. $\rightarrow$ 32-24 = 8  = $\blue{\mathbf{2\times 2}}\times 2$\only<13->{
                    \begin{tcolorbox}
                      $\mathbf{\underset{\text{Position 1 \texttt{IKB}}}{24}+\underset{\text{Codewort-Länge!}}{\blue{4}}\times \underset{\text{Anzahl Codewörter}}{2} = \underset{\text{Position 2 \texttt{IKB}}}{32}}$
                    \end{tcolorbox}
                  }
            \item<8-> \textbf{Könnte die Schlüssellänge auch 12 sein?} \only<10->{Nein, denn dann würden die beiden \texttt{IKB} nicht auf den selben Schlüsselteil fallen!}
            \item<11-> Distanz zwischen Positionen = Vielfaches der Schlüssellänge?
          \end{itemize}
  \end{itemize}
