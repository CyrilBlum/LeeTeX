\section{Classification vs. Prediction}

\begin{mydefinition}[Classification]
Bei der \textbf{Classification} wird ein Datenpunkt einer \textbf{Kategorie} zugeordnet, z.\,B. \enquote{bestanden / nicht bestanden}.
\end{mydefinition}

\begin{mydefinition}[Prediction (Regression)]
Bei der \textbf{Prediction} wird ein \textbf{Zahlenwert} vorhergesagt, z.\,B. eine erwartete Wohnungsmiete.
\end{mydefinition}

\begin{myexample}
\begin{itemize}
    \item Classification: Ist eine E-Mail Spam oder kein Spam?
    \item Prediction: Wie hoch ist der Stromverbrauch nächsten Monat?
\end{itemize}
\end{myexample}
