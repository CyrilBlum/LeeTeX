\begin{frame}{Motivation: Schlüssel-Wert-Paare}
	\begin{itemize}[<+->]
		\item Wie speichert man zu einem Namen eine Telefonnummer?
		\item Wie kann man zu einem Produkt den Preis speichern?
		\item Wie kann man zu einem Land die Hauptstadt speichern?
		\item Mit Listen: möglich, aber unpraktisch (z.B. Index-Suche)
		\item Lösung: \textbf{Dictionaries} (Wörterbücher)
	\end{itemize}
\end{frame}

\begin{frame}[fragile]{Was ist ein Dictionary?}
	\begin{itemize}[<+->]
		\item Eine Sammlung von \textbf{Schlüssel-Wert-Paaren}
		\item Zugriff nicht über Index, sondern über einen eindeutigen \textbf{Schlüssel} \emoji{key}
		\item \textbf{Beispiel}:
		      \begin{lstlisting}
telefonbuch = {
    "Anna": "079 123 45 67",
    "Clara": "077 555 44 33"
}
print(telefonbuch["Anna"]) # druckt 079 123 45 67
	\end{lstlisting}
	Schlüssel sind hier Namen \lstinline|"Anna"| und \lstinline|"Clara"|, Werte sind die Telefonnummern.
	\end{itemize}

\end{frame}

\begin{frame}[fragile]{Dictionary: Definition}
	\begin{block}{Definition}
		Ein \textbf{Dictionary} ist eine Datenstruktur, die jedem Schlüssel (\textit{key}) einen Wert (\textit{value}) zuordnet. Die Schlüssel müssen eindeutig sein und können z.B. Zahlen oder Zeichenketten sein.
	\end{block}
\end{frame}

\begin{frame}[fragile]{Beispiel: Produktpreise}
	\begin{lstlisting}
preise = {
    "Apfel": 0.80,
    "Brot": 2.50
}
print(preise["Brot"])  # gibt 2.5 aus
	\end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Werte hinzufügen und ändern}
	\begin{lstlisting}
# Dictionary erstellen
preise = {
    "Apfel": 0.80,
    "Brot": 2.50
}

preise["Milch"] = 1.60   # neues Produkt hinzufügen
preise["Apfel"] = 0.90   # Preis ändern

print(preise)
# gibt {'Apfel': 0.9, 'Brot': 2.5, 'Milch': 1.6} aus
	\end{lstlisting}
	\begin{center}
		Wie bei Listen, aber mit Schlüsseln!
	\end{center}
\end{frame}

\begin{frame}[fragile]{Alle Schlüssel und Werte durchgehen}
	\begin{lstlisting}[basicstyle=\ttfamily\footnotesize]
for produkt in preise:
    print(produkt, "kostet", preise[produkt], "Franken")
	\end{lstlisting}
	\begin{itemize}[<+->]
		\item \lstinline|for produkt in preise| geht durch alle \textbf{Schlüssel} \emoji{key}
		\item \lstinline|preise[produkt]| gibt den Wert zum Schlüssel zurück
	\end{itemize}
\end{frame}
