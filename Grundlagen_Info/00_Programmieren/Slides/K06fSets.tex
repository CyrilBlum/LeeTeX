\begin{frame}[fragile]{Listen zu Mengen umwandeln und Mengenoperationen}
    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\footnotesize]
liste_a = ["Anna", "Ben", "Clara"]
liste_b = ["Ben", "David", "Eva"]

# Listen zu Mengen umwandeln
set_a = set(liste_a)
set_b = set(liste_b)

# Mengenoperationen
vereinigung = set_a | set_b   # Alle Namen aus beiden Listen
schnittmenge = set_a & set_b  # Namen in beiden Listen
differenz = set_a - set_b     # Namen nur in liste_a

print("Vereinigung:", vereinigung)
print("Schnittmenge:", schnittmenge)
print("Differenz:", differenz)
    \end{lstlisting}
    \begin{itemize}
        \item Mit \lstinline|set(liste)| wird eine Liste in eine Menge umgewandelt.
        \item \textbf{Mengenoperationen}: \lstinline||| (=Vereinigung), \lstinline|&| (=Schnittmenge), \lstinline|-| (=Schnittmenge)
    \end{itemize}
\end{frame}
