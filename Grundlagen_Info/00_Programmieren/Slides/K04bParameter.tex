\begin{frame}[fragile]{Lebensdauer von Variablen (Scope)}
    \begin{columns}
        \begin{column}{.5\textwidth}
            \begin{lstlisting}
def meine_funktion(x):
    y = x + 10
    print(y)

meine_funktion(5)
print(y)  # Fehler!
\end{lstlisting}
            \vspace{.3cm}
            \textcolor{red}{NameError: name 'y' is not defined}
        \end{column}
        \begin{column}{.5\textwidth}
            \begin{itemize}[<+->]
                \item \textbf{Parameter} und lokal definierte Variablen existieren nur \textbf{innerhalb} der Funktion
                \item Ausserhalb der Funktion sind sie \textbf{nicht sichtbar}
                \item Dies nennt man \tib{Scope} (Gültigkeitsbereich)
                \item \textbf{Vorteil}: Variablennamen können mehrfach verwendet werden
                \item \textbf{Lebensdauer} muss beachtet werden: Sobald die Funktion verlassen wird, existieren die Parameter und lokalen Variablen nicht mehr
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}



\begin{frame}[fragile]{Variablen}
    Was sind \textbf{Parameter}?
    \begin{itemize}[<+->]
        \item \textbf{Parameter} sind Variablen
        \item Ein Wort, das einen \tib{veränderbaren} (variablen) Wert enthält.
        \item Variablenname darf (fast) frei gewählt werden
              \begin{itemize}
                  \item Name muss mit einem Buchstaben oder Unterstrich beginnen (nicht: \verb|1meinname|)
                  \item Kein von Python reserviertes Wort (\lstinline|def, if, for| etc.)
                  \item Falls zusammengesetzter Name: Trennen Sie die Teile durch Unterstriche (z.B.: \lstinline|mein_name|)
                  \item \textbf{Sinnvoller} und \textbf{beschreibender} Name!
              \end{itemize}
    \end{itemize}
\end{frame}
