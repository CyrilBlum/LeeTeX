import turtle as t
t.hideturtle()

def quadrat():
    for _ in range(4):
        t.forward(100)
        t.right(90)

for _ in range(10):
    t.color("red")
    quadrat()
    t.delay(500)
    t.color("black")
    quadrat()
    t.delay(500)

t.done()
