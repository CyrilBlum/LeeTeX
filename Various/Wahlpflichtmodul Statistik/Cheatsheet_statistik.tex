\textcolor{\cheatSheetColor}{\subsection*{Skalenniveaus}}

\begin{cheatSheetBox}[title=Variable]
Eine Variable ist eine Eigenschaft,
die bei verschiedenen Personen unterschiedliche Werte annehmen kann.

Beispiel:
Schlafdauer in Stunden.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Nominalskala]
Kategorien ohne Ordnung.

Beispiele: 

\begin{enumerate*} \item Geschlecht \item Nationalität \item Lieblingsmusik \end{enumerate*}

Sinnvolle Kennzahlen: 

\begin{enumerate*} \item Häufigkeit \item Modus \end{enumerate*}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Ordinalskala]
Kategorien mit Rangordnung.

Beispiele:

\begin{enumerate*}
\item Schulnoten
\item Ranglisten
\item Zufriedenheitsskala
\end{enumerate*}

Sinnvolle Kennzahlen:

\begin{enumerate*}
\item Median
\item Quartile
\end{enumerate*}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Intervallskala]
Gleiche Abstände, aber kein echter Nullpunkt.

Beispiel: Temperatur in $^{\circ}$C

Erlaubt:

\begin{enumerate*}
\item Mittelwert
\item Standardabweichung
\end{enumerate*}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Verhältnisskala]
Numerische Werte mit echtem Nullpunkt.

Beispiele:

\begin{enumerate*}
\item Gewicht
\item Länge
\item Einkommen
\end{enumerate*}

Verhältnisse sind sinnvoll.
\end{cheatSheetBox}

\textcolor{\cheatSheetColor}{\subsection*{Lagekennzahlen}}

\begin{cheatSheetBox}[title=Minimum und Maximum]
Kleinster und grösster Wert im Datensatz.

Spannweite:
\[
\text{Spannweite} = \text{Max} - \text{Min}
\]
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Modus]
Häufigster Wert im Datensatz.

Geeignet für alle Skalenniveaus, besonders Nominalskala.

Beispiel: 1, 2, 2, 3, 4
Modus = 2
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Mittelwert]
Durchschnitt aller Werte.

\[
\bar{x} = \frac{1}{n}\sum_{i=1}^{n} x_i
\]

Empfindlich gegenüber Ausreissern.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Median]
Mittlerer Wert nach Sortierung.

Robust gegenüber Ausreissern.

Beispiel:

1, 2, 3, 4, 100

Median = 3
\end{cheatSheetBox}
\textcolor{\cheatSheetColor}{\subsection*{Streuungsmasse}}

\begin{cheatSheetBox}[title=Quartile]
Datensatz wird in vier Teile geteilt.

\begin{itemize}
\item Q1 = 25\%
\item Median = 50\%
\item Q3 = 75\%
\end{itemize}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Quartilsabstand]
Interquartilsabstand:

\[
\gls{IQR} = Q3 - Q1
\]

Robust gegenüber Ausreissern.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Standardabweichung]

Misst Streuung um den Mittelwert.

\[
s = \sqrt{\frac{1}{n-1}\sum (x_i-\bar{x})^2}
\]

\textbf{Empirische Regel:}
\begin{itemize}
\item $\pm$ 1 \gls{SD}: ca. 68\%
\item $\pm$ 2 \gls{SD}: ca. 95\%
\item $\pm$ 3 \gls{SD}: ca. 99.7\%
\end{itemize}
\end{cheatSheetBox}
\columnbreak
\textcolor{\cheatSheetColor}{\subsection*{Ausreisser und Verteilungen}}

\begin{cheatSheetBox}[title=Ausreisser]
Wert, der deutlich vom Rest der Daten abweicht.

Faustregel (Boxplot):
\begin{itemize}
\item $< Q1 - 1.5 \cdot$ \gls{IQR}
\item $> Q3 + 1.5 \cdot$ \gls{IQR}
\end{itemize}

Behandlung:
\begin{enumerate*}
\item Median statt Mittelwert
\item \gls{IQR} statt \gls{SD}
\end{enumerate*}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Verteilungsformen]

\textbf{Symmetrisch (unimodal):}
Mittelwert $\sim$ Median

\textbf{Rechtsschief:}
Wenige hohe Ausreisser,
Mittelwert $>$ Median

\textbf{Linksschief:}
Wenige niedrige Ausreisser,
Mittelwert $<$ Median

\textbf{Bimodal:}
Zwei Häufigkeitsgipfel
$\rightarrow$ Hinweis auf zwei Gruppen
\end{cheatSheetBox}

\textcolor{\cheatSheetColor}{\subsection*{Gute statistische Fragen}}

\begin{cheatSheetBox}[title=Von Thema zu Statistikfrage]
Weniger geeignet:
\enquote{Wer ist der:die Grösste?}

Statistisch geeignet:
\enquote{Wie ist die Verteilung der Körpergrössen?}

Faustregel:
Nicht einzelne Personen, sondern Merkmale (Variablen) untersuchen.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Vier gute Fragetypen]
\begin{enumerate}
\item Verteilung: Wie häufig? Was ist typisch?
\item Vergleich: Unterschiede zwischen Gruppen?
\item Beziehung: Hängen zwei Variablen zusammen?
\item Vorhersage: Welche Tendenz zeigt der Datensatz?
\end{enumerate}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Präzise formulieren]
Eine gute Frage nennt klar:
\begin{enumerate*}
\item Zielgruppe
\item Variable
\item Vergleich oder Bezug
\end{enumerate*}
\end{cheatSheetBox}

\textcolor{\cheatSheetColor}{\subsection*{Schliessende Statistik}}

\begin{cheatSheetBox}[title=Stichprobe und Grundgesamtheit]
Stichprobe:
beobachtete Teilmenge.

Grundgesamtheit:
grössere Gruppe, über die eine Aussage gemacht wird.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Repräsentativ und signifikant]
\textbf{Repräsentativ:}
Die Stichprobe bildet die Zielgruppe ausreichend gut ab.

\textbf{Signifikant:}
Ein beobachteter Unterschied ist statistisch nicht gut durch Zufall erklärbar.

Signifikant bedeutet nicht automatisch inhaltlich wichtig.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Vertrauensintervall (95\%)]
Ein Vertrauensintervall gibt einen plausiblen Bereich für einen unbekannten Wert an.

Lesart:
\begin{enumerate*}
\item schmaler = präziser
\item breiter = unsicherer
\end{enumerate*}

Wenn 0 im Intervall eines Unterschieds liegt,
ist kein klarer Unterschied nachgewiesen.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Kontrollgruppe]
Eine Kontrollgruppe liefert den Referenzwert ohne Behandlung.
Sie ist zentral für faire Vergleiche und vorsichtige Interpretation.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Einfluss von n]
Bei grösserer Stichprobe $n$ wird das Vertrauensintervall in der Regel enger.

Folge:
Die Aussage wird präziser.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Symbolübersicht (Gähn-Beispiel)]
$n$: Anzahl Personen pro Gruppe.

$x$: Anzahl mit Ereignis (z.B. gähnen).

$\hat{p}=x/n$: beobachteter Anteil.

$\hat{\Delta}$: Unterschied der beobachteten Anteile.

$\mathrm{SE}(\hat{\Delta})$: zufallsbedingte Streuung dieses Unterschieds.
\end{cheatSheetBox}

\textcolor{\cheatSheetColor}{\subsection*{Interpretation und Kritik}}

\begin{cheatSheetBox}[title=Kontingenztabelle]
Erlaubt:
beschreibende Aussagen zur Stichprobe.

Nicht erlaubt:
kausale Aussagen ohne geeignetes Design.
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Grafiken kritisch lesen]
Prüfen:
\begin{enumerate}
\item Startet die Achse bei 0?
\item Sind Kategorien gleich breit?
\item Fehlen Gruppen oder Werte?
\item Passen Aussage und Datenlage zusammen?
\end{enumerate}
\end{cheatSheetBox}

\begin{cheatSheetBox}[title=Arbeitsprozess in 4 Schritten]
\begin{enumerate}
\item Fragestellung klären
\item Daten beschreiben
\item Unsicherheit beurteilen
\item Aussagebereich reflektieren
\end{enumerate}
\end{cheatSheetBox}
