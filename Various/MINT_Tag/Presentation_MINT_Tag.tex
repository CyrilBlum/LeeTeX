\begin{frame}{1. Erstellung von Unterrichtsmaterialien}
	Anforderungen an unsere Vorgehensweise:
	\begin{itemize}
		\item sehr saubere Darstellung \pause
		\item Unser \LaTeX{}-Setup soll eine \enquote{Eierlegende Wollmilchsau} sein: Erstellung verschiedenster Dokumentenklassen mit einem einzigen Setup (e.g. Skripte, Slides, Prüfungen, $\ldots$). \pause
		\item nahtlose Kooperation (Version-Control) über GitHub \pause
		\item Ermöglichung eines effizienten Workflows \pause
	\end{itemize}
	$\rightarrow$ Demonstration
\end{frame}

\begin{frame}{2. Lernplattform: Moodle}
	\begin{itemize}
		\item die gesamte Fachschaft Informatik verwendet, neben Moodle, keine weitere Lernplattform \pause
		\item Plattform zur Ablage von Materialien, Abgabe von Lösungen durch die SuS, Prüfungen, Übungsbetrieb \pause
		\item Einladung von anderen Lehrpersonen in Moodle-Kurse
	\end{itemize}
\end{frame}


\begin{frame}{3. Übungsaufgaben und Prüfungen}
	\begin{itemize}
		\item für Prüfungen: Safe Exam Browser (konfigurieren durch \textit{SEB Configuration Tool}) \pause
		\item gemeinsame Aufgabensammlung \pause
		\item \textit{CodeRunner} erlaubt automatisiertes Prüfen von Code \pause
		\item \textit{STACK} erlaubt die Erstellung komplexer Fragen und eine algebraische Auswertung (e.g. das Prüfen auf algebraische gleichheit) \pause
		\item Jahrgangsprüfung Mathematik \pause
		\item verwende die Hilfe von LLMs für STACK-Fragen
	\end{itemize}
\end{frame}

\begin{frame}{4. Zukunftspläne}
	\begin{itemize}
		\item deployment pipeline (CI / CD) soll Dokumente auf Moodle automatisch durch neu kompilierte \LaTeX{}-Dokumente ersetzen
		\item Korrektur durch LLMs?
	\end{itemize}
\end{frame}
